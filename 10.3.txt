// Дано целое число N (> 0). Найти квадрат данного числа, используя для его вычисления следующую формулу: N2 = 1 + 3 + 5 + . . . + (2·N ? 1). После добавления к сумме каждого слагаемого выводить текущее значение суммы
#include <iostream>
#include <string>
using namespace std;
int main()
{
  int N;
  float i,x;
  cout << "N = ";
  cin >> N;
  if (N > 0)
  {
  x=0;
    for (i=1; i<=N; i=i+1)
        {
          x=x+(2*i-1);
          cout << "Sum"<< i <<" = " << x << "\n";
        }
    cout << "N^2 = " << x;
  }
  else
  {
      cout << "Error" ;
  }
}
