// Дано целое число N (> 0). Найти произведение 1.1 · 1.2 · 1.3 · . . . (N сомножителей).
#include <iostream>
#include <string>
using namespace std;
int main()
{
  int N;
  float i,x;
  cout << "N = ";
  cin >> N;
  if (N > 0)
  {
  x=1;
    for (i=1; i<=N; i=i+1)
        {
          x=x*(1+i/10);
        }
    cout << "Proizvedenie = " << x;
  }
  else
  {
      cout << "Error" ;
  }
}
